\definecolor{vsmag}{RGB}{196, 72, 255} 
\definecolor{vsblue}{RGB}{0, 0, 255}
\definecolor{vsgreen}{RGB}{52, 128, 0}
\definecolor{vsred}{RGB}{163, 21, 21}
\definecolor{vsgray}{RGB}{128, 128, 128}

\colorlet{keywordvar}{vsblue}
\colorlet{keywordcontrol}{vsmag}
\colorlet{comment}{vsgreen}
\colorlet{string}{vsred}

\lstdefinelanguage{CVS}{                          % Custom syntax highlighting
	basicstyle   = \footnotesize \ttfamily,
	alsoletter = {\#},
	morecomment=[l]{//}, %use comment-line-style!
	morecomment=[s]{/*}{*/}, %for multiline comments
	morekeywords=[1]{                              % All Keywords in color blue
		abstract, event, new, struct, explicit, extern, this, bool, false, operator, true, override, float, private, char, protected, public, class, const,
		using, int, virtual, interface, sealed, volatile, delegate, internal, short, void, sizeof, double, long, static,
		enum, namespace
	},
	morekeywords=[2]{                              % All Keywords in color red
		switch, throw, break, finally, case, try, params, typeof,
		catch, for, goto, if, continue, return, default, do,  while, else
	},
	morekeywords=[3]{                              % All Keywords in color red
	\#define, \#include
	},
	keywordstyle = [1]\color{keywordvar}\bfseries,
	keywordstyle = [2]\color{keywordcontrol}\bfseries,
	keywordstyle = [3]\color{vsgray}\bfseries,
	commentstyle = \color{comment},                            % All Keywords in color green
	stringstyle = \color{string}\ttfamily,
	morestring=[b]',
	morestring=[b]"
}

\lstdefinestyle{cvs}{
    backgroundcolor=\color{lightgray},
    frame        = single, 
    language     = CVS,
    breaklines   = true,                           % sets automatic line breaking
    tabsize      = 2,                              % sets default tabsize to 3 spaces
    numbers=left, 
    numberstyle=\tiny, 
    numbersep=5pt,  
    extendedchars=true,
    captionpos=b 
}
